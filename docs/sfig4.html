23.6 E: correlation stemness & cytotrace

pt <- FetchData(srt_epi, vars = c("CytoTRACE2_Relative_Score", "CD44", "SOX2", "BMI1"), layer = "data")
pt <- filter(pt, CD44 > 0, SOX2 > 0, BMI1 > 0)
pt <- pt %>% tidyr::pivot_longer(cols = c(CD44, SOX2, BMI1), names_to = "gene", values_to = "exp") %>% 
  mutate(CytoTRACE2_Relative_Score = if_else(CytoTRACE2_Relative_Score > 0.75, 0.75, CytoTRACE2_Relative_Score))

x_range <- range(pt$CytoTRACE2_Relative_Score)
x_seq <- seq(x_range[1], x_range[2], length.out = 100)

predictions <- data.frame()
for(g in unique(pt$gene)) {
  subset_data <- pt[pt$gene == g, ]
  loess_fit <- loess(exp ~ CytoTRACE2_Relative_Score, data = subset_data)
  y_pred <- predict(loess_fit, newdata = data.frame(CytoTRACE2_Relative_Score = x_seq))
  temp_df <- data.frame(
    CytoTRACE2_Relative_Score = x_seq,
    exp = y_pred,
    gene = g
  )
  
  predictions <- rbind(predictions, temp_df)
}

p_normalized <- ggplot(predictions, aes(x = CytoTRACE2_Relative_Score, y = exp, color = gene)) + 
  geom_line(size = 1.2) + 
  scale_color_discrete(name = "Gene") +
  theme_classic() +
  theme(axis.line.x = element_blank(),
        axis.ticks.x = element_blank(),
        axis.text.x = element_blank(),
        panel.grid = element_blank()
  )

color_bar_data <- data.frame(
  CytoTRACE2_Relative_Score = seq(x_range[1], x_range[2], length.out = 100),
  y = 1
)
color_bar <- ggplot(color_bar_data, aes(x = CytoTRACE2_Relative_Score, y = y, fill = CytoTRACE2_Relative_Score)) +
  geom_tile(height = 0.6) +
  scale_fill_gradientn(
    colors = c("#431954", "#6EC162")
  ) + 
  theme(
    axis.title.y = element_blank(), 
    axis.text.y = element_blank(),
    axis.ticks.y = element_blank(), 
    legend.position="none"
    )

combined_plot <- plot_grid(
  p_normalized,
  color_bar,
  ncol = 1,
  rel_heights = c(6, 1), 
  align = "v",
  axis = "lr"
)

ggsave(glue("{fig_dir}/correlation_stemness_cytotrace_norm.pdf"), combined_plot, width = 6, height = 5)
ggsave(glue("{img_dir}/correlation_stemness_cytotrace_norm.png"), combined_plot, width = 6, height = 5)

23.8 G: k3 dendrogram

pt <- FetchData(srt_epi, vars = c("CD44", "sample_id"), layer = "data") %>% 
  dplyr::filter(sample_id != "P05", sample_id != "P46") # filter sample CD44 no exp

# ks test & hclust
sps <- unique(pt$sample_id)
n <- length(sps)
ks_dist <- matrix(0, nrow = n, ncol = n)
rownames(ks_dist) <- sps
colnames(ks_dist) <- sps

for (i in 1:(n - 1)) {
  for (j in (i + 1):n) {
    ks_test <- ks.test(pt[pt$sample_id == sps[i], "CD44"], pt[pt$sample_id == sps[j], "CD44"])
    ks_dist[i, j] <- ks_test$statistic  
    ks_dist[j, i] <- ks_dist[i, j]
  }
}

dist_mat <- as.dist(ks_dist)
hc <- hclust(dist_mat, method = "ward.D2")
sp_order <- hc$labels[hc$order]

clusters <- cutree(hc, k = 3)
df_cl <- data.frame(sample_id = names(clusters),
                    cluster = clusters) %>%
  mutate(order = match(sample_id, sp_order))
fwrite(df_cl, glue("{fig_dir}/density_CD44_sample_hclust_k3.csv"))

# density ridges colored by cluster
df_cl <- fread(glue("{fig_dir}/density_CD44_sample_hclust_k3.csv")) %>%
  arrange(order)
pt <- FetchData(srt_epi, vars = c("CD44", "sample_id"), layer = "data") %>% 
  dplyr::filter(sample_id != "P05", sample_id != "P46") %>% 
  left_join(df_cl, by = "sample_id") %>%
  mutate(sample_id = factor(sample_id, levels = df_cl$sample_id))
p <- ggplot(pt, aes(CD44, sample_id)) +
  geom_density_ridges(aes(fill = as.factor(cluster))) +
  scale_fill_manual(values = cols_cl) +
  labs(fill = "cluster") +
  theme_classic()
ggsave(glue("{fig_dir}/density_CD44_sample_cluster_k3.pdf"), p, width = 6, height = 15)
ggsave(glue("{img_dir}/density_CD44_sample_cluster_k3.png"), p, width = 6, height = 15)
pdf(glue("{fig_dir}/density_CD44_sample_cluster_k3_dendrogram.pdf"), width = 12, height = 6)
plot(rev(as.dendrogram(hc)))
dev.off()
png(glue("{img_dir}/density_CD44_sample_cluster_k3_dendrogram.png"), width = 12, height = 6, units = "in", res = 300)
plot(rev(as.dendrogram(hc)))
dev.off()

23.9 H: KEGG enrichment k3

# barplot of selected pathway
df <- readxl::read_xlsx(glue("{dat_dir}/CD44_cluster_markers_enrichment_KEGG_k3_selected.xlsx"))
pt <- df %>%
  arrange(cluster, p.adjust) %>%
  mutate(p.adjust = as.numeric(p.adjust))
pmax <- max(-log10(pt$p.adjust)) %>% ceiling
p_lst <- lapply(unique(pt$cluster), function(x){
  pt1 <- pt %>%
    filter(cluster == x) %>%
    mutate(Description = factor(Description, levels = Description))
  ymax <- max(-log10(pt1$p.adjust)) %>% floor
  p <- ggplot(pt1, aes(Description, -log10(p.adjust))) +
    geom_bar(stat = "identity", fill = unname(cols_cl)[x], width = 0.6) +
    ylim(0, pmax) +
    coord_flip() +
    geom_hline(yintercept = 1:ymax, color = "white") +
    theme_classic() +
    labs(x = paste0("Cluster ", x))
  return(p)
})
hts <- pt %>% count(cluster) %>% pull(n)
p <- wrap_plots(p_lst, ncol = 1, heights = hts)
ggsave(glue("{fig_dir}/CD44_cluster_markers_enrichment_KEGG_k3_poi_barplot.pdf"), p, width = 8, height = 8)
ggsave(glue("{img_dir}/CD44_cluster_markers_enrichment_KEGG_k3_poi_barplot.png"), p, width = 8, height = 8)
